001package org.intellimate.izou.sdk.frameworks.music.resources;
002
003import org.intellimate.izou.identification.Identification;
004import org.intellimate.izou.sdk.frameworks.music.player.Playlist;
005import org.intellimate.izou.sdk.resource.Resource;
006
007import java.util.HashMap;
008
009/**
010 * use this method obtain a specified Playlist.
011 * Internally it keeps the PlaylistData in the HashMap when the request happens, the key is the ResourceID.
012 * @author LeanderK
013 * @version 1.0
014 */
015public class BroadcasterPlaylist extends Resource<HashMap<String, Object>> {
016    @SuppressWarnings("SpellCheckingInspection")
017    public static final String RESOURCE_ID = "izou.music.resource.broadcaster.playlist";
018
019    /**
020     * creates a new Resource.
021     *
022     * @param provider   the Provider of the Resource
023     */
024    public BroadcasterPlaylist(Identification provider) {
025        super(RESOURCE_ID, provider);
026    }
027
028    /**
029     * creates a new Resource.
030     *
031     * @param provider   the Provider of the Resource
032     * @param hashMap    the resource
033     */
034    protected BroadcasterPlaylist(Identification provider, HashMap<String, Object> hashMap) {
035        super(RESOURCE_ID, provider, hashMap);
036    }
037
038    /**
039     * creates a new Resource.
040     *
041     * @param consumer   the ID of the Consumer
042     * @param hashMap    the resource
043     */
044    protected BroadcasterPlaylist(HashMap<String, Object> hashMap, Identification consumer) {
045        super(RESOURCE_ID, hashMap, consumer);
046    }
047
048    /**
049     * creates a new Resource.
050     * This method is thread-safe.
051     *
052     * @param provider   the Provider of the Resource
053     * @param consumer   the ID of the Consumer
054     */
055    protected BroadcasterPlaylist(Identification provider, Identification consumer) {
056        super(RESOURCE_ID, provider, consumer);
057    }
058
059    /**
060     * creates a new Resource.
061     *
062     * @param provider   the Provider of the Resource
063     * @param hashMap    the resource
064     * @param consumer   the ID of the Consumer
065     */
066    protected BroadcasterPlaylist(Identification provider, HashMap<String, Object> hashMap, Identification consumer) {
067        super(RESOURCE_ID, provider, hashMap, consumer);
068    }
069
070    /**
071     * creates the Playlist-Request
072     * @param consumer the consumer who wants to ask for the playlist
073     * @param playlistName the name of the playlist
074     * @return the resource
075     */
076    public static BroadcasterPlaylist createPlaylistRequest(Identification consumer, String playlistName) {
077        HashMap<String, Object> hashMap = new HashMap<>();
078        hashMap.put(RESOURCE_ID, playlistName);
079        return new BroadcasterPlaylist(hashMap, consumer);
080    }
081
082    /**
083     * creates the Playlist-Answer
084     * @param provider the provider
085     * @param playlist the playlist
086     * @return the resource
087     */
088    public static BroadcasterPlaylist createPlaylistAnswer(Identification provider, Playlist playlist) {
089        return new BroadcasterPlaylist(provider, playlist.export());
090    }
091}
