027    public static final String idDescriptor = "izou.music.trackinfo.data";
028    private final String data;

119    /**
120     * retruns true if any of these arguments is already existing and would be overwritten, otherwise retruns false.
121     * The method also returns true if everything is null except the albumCover-Format (maybe known in advance).
122     * @param trackInfo the trackInfo to compare against
123     * @return true if new
124     */
125    public boolean isNew(TrackInfo trackInfo) {
126        return isNew(trackInfo.name,
127                trackInfo.artist,
128                trackInfo.album,
129                trackInfo.albumCover,
130                trackInfo.albumCoverFormat,
131                trackInfo.data);
132    }
133
134    /**
135     * retruns true if any of these arguments is already existing and would be overwritten, otherwise retruns false.
136     * The method also returns true if everything is null except the albumCover-Format (maybe known in advance).
137     * @param name the name
138     * @param artist the artist
139     * @param album the album
140     * @param albumCover the album cover
141     * @param albumCoverFormat the album cover format
142     * @param id the id of the track
143     * @return true if new
144     */
145    public boolean isNew(String name, String artist, String album, byte[] albumCover, String albumCoverFormat, String id) {
146        if (name == null && artist == null && album == null && albumCover == null && id == null)
147            return true;
148        BiPredicate<String, String> compareStrings = (newString, oldString) ->
149                newString != null && (oldString == null || !newString.equals(oldString));
150        if (compareStrings.test(name, this.name)) {
151            return true;
152        }
153        if (compareStrings.test(artist, this.artist)) {
154            return true;
155        }
156        if (compareStrings.test(album, this.album)) {
157            return true;
158        }
159        if (compareStrings.test(id, this.data)) {
160            return true;
161        }
162        if (compareStrings.test(albumCoverFormat, this.albumCoverFormat)) {
163            return true;
164        }
165        if (albumCover != null) {
166            if (this.albumCover == null || !Arrays.equals(albumCover, this.albumCover))
167                return true;
168        }
169        return false;
170    }
171
172    /**
173     * returns a trackinfo if some information was added and not overwritten
174     * @param trackInfo the data to compare
175     * @return a trackinfo if some information was updated
176     */
177    public Optional<TrackInfo> update(TrackInfo trackInfo) {
178        return update(trackInfo.name,
179                trackInfo.artist,
180                trackInfo.album,
181                trackInfo.albumCover,
182                trackInfo.albumCoverFormat,
183                trackInfo.data);
184    }
185
186    /**
187     * returns a trackinfo if some information was added and not overwritten (see isNew) AND a change occurred.
188     * @param name the name
189     * @param artist the artist
190     * @param album the album
191     * @param albumCover the album cover
192     * @param albumCoverFormat the album cover format
193     * @param id the id of the track
194     * @return a trackInfo if some information was updated, and an empty optional if: 1. information would be ovewritten
195     * or 2. no change occured
196     */
197    public Optional<TrackInfo> update(String name, String artist, String album, byte[] albumCover, String albumCoverFormat, String id) {
198        if (isNew(name, artist, album, albumCover, albumCoverFormat, id))
199            return Optional.empty();
200        boolean change = false;
201        if (name != null && !name.equals(this.name)) {
202            change = true;
203        }
204        if (artist != null && !artist.equals(this.artist)) {
205            change = true;
206        }
207        if (album != null && !album.equals(this.album)) {
208            change = true;
209        }
210        if (name != null && !name.equals(this.name)) {
211            change = true;
212        }
213        if (albumCoverFormat != null && !albumCoverFormat.equals(this.albumCoverFormat)) {
214            change = true;
215        }
216        if (id != null && !id.equals(this.data)) {
217            change = true;
218        }
219        if (!change)
220            return Optional.empty();
221        return Optional.of(new TrackInfo(
222                this.name == null? name : this.name,
223                this.artist == null? artist : this.artist,
224                this.album == null? album : this.album,
225                this.albumCover == null? albumCover : this.albumCover,
226                this.albumCoverFormat == null? albumCoverFormat : this.albumCoverFormat,
227                this.data == null? id : this.data
228        ));
229    }

246    /**
247     * returns the optional TrackInfo if the HashMap contains no malformed data
248     * @param hashMap the data to import from
249     * @return an optional
250     */
251    public static Optional<TrackInfo> importFromHashMap(HashMap<String, Object> hashMap) {
252        try {
253            String name = (String) hashMap.get(nameDescriptor);
254            String album = (String) hashMap.get(albumDescriptor);
255            String artist = (String) hashMap.get(artistDescriptor);
256            byte[] albumCover = (byte[]) hashMap.get(albumCoverDescriptor);
257            String albumCoverFormat = (String) hashMap.get(albumCoverFormatDescriptor);
258            String id = (String) hashMap.get(idDescriptor);
259            return Optional.of(new TrackInfo(name, artist, album, albumCover, albumCoverFormat, id));
260        } catch (ClassCastException e) {
261            return Optional.empty();
262        }
263    }

265    /**
266     * creates a TrackInfo from the resourceModel
267     * @param resourceModel the resourceModel
268     * @return the optional TrackInfo
269     */
270    public static Optional<TrackInfo> importFromResource(ResourceModel resourceModel) {
271        Object resource = resourceModel.getResource();
272        try {
273            @SuppressWarnings("unchecked")
274            HashMap<String, Object> hashMap = (HashMap<String, Object>) resource;
275            return importFromHashMap(hashMap);
276        } catch (ClassCastException e) {
277            return Optional.empty();
278        }
279    }

281    @Override
282    public boolean equals(Object o) {
283        if (this == o) return true;
284        if (!(o instanceof TrackInfo)) return false;
285
286        TrackInfo trackInfo = (TrackInfo) o;
287
288        if (name != null ? !name.equals(trackInfo.name) : trackInfo.name != null) return false;
289        if (artist != null ? !artist.equals(trackInfo.artist) : trackInfo.artist != null) return false;
290        if (album != null ? !album.equals(trackInfo.album) : trackInfo.album != null) return false;
291        if (!Arrays.equals(albumCover, trackInfo.albumCover)) return false;
292        if (albumCoverFormat != null ? !albumCoverFormat.equals(trackInfo.albumCoverFormat) : trackInfo.albumCoverFormat != null)
293            return false;
294        return !(data != null ? !data.equals(trackInfo.data) : trackInfo.data != null);
295
296    }

298    @Override
299    public int hashCode() {
300        int result = name != null ? name.hashCode() : 0;
301        result = 31 * result + (artist != null ? artist.hashCode() : 0);
302        result = 31 * result + (album != null ? album.hashCode() : 0);
303        result = 31 * result + (albumCover != null ? Arrays.hashCode(albumCover) : 0);
304        result = 31 * result + (albumCoverFormat != null ? albumCoverFormat.hashCode() : 0);
305        result = 31 * result + (data != null ? data.hashCode() : 0);
306        return result;
307    }
