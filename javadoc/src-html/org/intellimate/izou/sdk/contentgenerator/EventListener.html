001package org.intellimate.izou.sdk.contentgenerator;
002
003import org.intellimate.izou.events.EventModel;
004import org.intellimate.izou.identification.Identifiable;
005import org.intellimate.izou.identification.IdentificationManager;
006import org.intellimate.izou.sdk.events.CommonEvents;
007import org.intellimate.izou.sdk.events.Event;
008
009import java.util.Collections;
010import java.util.Optional;
011
012/**
013 * This class is used in the ContentGenerator to signal that a ContentGenerator wants to get triggered when it gets
014 * fired.
015 * @author Leander Kurscheidt
016 * @version 1.0
017 */
018public class EventListener {
019    private final Event event;
020    private final String descriptor;
021    private final String description;
022    private final String descriptorID;
023
024    /**
025     * Creates a new EventListener.
026     *
027     * @param event the Event to associate with
028     * @param descriptor the descriptor of the Event you want to listen to
029     * @param description the description of the descriptor
030     * @param descriptorID an ID for the descriptor (Should contain no special characters, spaces etc.)
031     */
032    EventListener(Event event, String descriptor, String description, String descriptorID) {
033        this.event = event;
034        this.descriptor = descriptor;
035        this.description = description;
036        this.descriptorID = descriptorID;
037    }
038
039    /**
040     * create the EventListener
041     * @param descriptor the descriptor of the Event you want to listen to
042     * @param description the description of the descriptor
043     * @param descriptorID an ID for the descriptor (Should contain no special characters, spaces etc.). Only String
044     *                     which match the regex (\w-_)+ are allowed.
045     * @param identifiable the identifiable which wants to create the EventListener
046     * @return if one of the parameter is null, or unable to to obtain ID
047     * @throws java.lang.IllegalArgumentException if the descriptorID contains illegal characters
048     */
049    public static Optional<EventListener> createEventListener(String descriptor, String description,
050                                                       String descriptorID, Identifiable identifiable)
051                                                                                        throws IllegalArgumentException{
052
053        if (!descriptorID.matches("[\\w\\-_]+"))
054            throw new IllegalArgumentException("descriptorID: " + descriptorID + " contains illegal characters");
055        return IdentificationManager.getInstance().getIdentification(identifiable)
056                .flatMap(id -> Event.createEvent(CommonEvents.Type.NOTIFICATION_TYPE, id, Collections.singletonList(descriptor)))
057                .map(event -> new EventListener(event, descriptor, description, descriptorID));
058    }
059
060    /**
061     * Returns the associated Event
062     *
063     * @return The Event
064     */
065    public EventModel<?> getEvent() {
066        return event;
067    }
068
069    /**
070     * Returns the descriptor of the Event you want to listen to
071     * @return a String
072     */
073    public String getDescriptor() {
074        return descriptor;
075    }
076
077    /**
078     * the description of the descriptor
079     * @return a String
080     */
081    public String getDescription() {
082        return description;
083    }
084
085    /**
086     * an ID for the descriptor (Should contain no special characters, spaces etc.)
087     * @return a String
088     */
089    public String getDescriptorID() {
090        return descriptorID;
091    }
092}
