013/**
014 * Manages property files, and is also a {@link ReloadableFile}
015 *
016 * <p>Unlike most manager classes in Izou, the PropertiesManager is included in every {@code AddOn} instance</p>
017 */
018public class PropertiesAssistant extends AddOnModule implements ReloadableFile {
019    private String propertiesPath;
020    private String defaultPropertiesPath;
021    private Properties properties;
022    private final EventPropertiesAssistant assistant;
023    private File propertiesFile;
024    private List<WeakReference<Consumer<PropertiesAssistant>>> listeners =
025            Collections.synchronizedList(new ArrayList<>());
026
027    public PropertiesAssistant(Context context, String addOnID) {
028        super(context, addOnID + ".PropertiesAssistant");
029        this.properties = new Properties();
030        this.propertiesPath = null;
031        this.defaultPropertiesPath = null;
032        this.assistant = new EventPropertiesAssistant(context, addOnID + ".EventPropertiesAssistant");
033        PluginWrapper plugin = getContext().getAddOn().getPlugin();
034        if (plugin != null) {
035            this.defaultPropertiesPath = getContext().getFiles().getLibLocation() +
036                    getContext().getAddOn().getPlugin().getPluginPath() + File.separator + "classes" + File.separator
037                    + "default_properties.txt";
038        } else {
039            //if we are debugging
040            this.defaultPropertiesPath = getContext().getAddOn().getClass().getClassLoader().
041                    getResource("default_properties.txt").getFile();
042        }
043        initProperties();
044    }
045
046    /**
047     * Gets the EventPropertiesAssistant
048     *
049     * @return the EventPropertiesAssistant

055    /**

077     * @return the value in this property list with the specified key value.
078     */

098     * @param key the key to be placed into this property list.
099     * @param value the value corresponding to key.

101    public void setProperty(String key, String value) {
102        properties.setProperty(key, value);
103    }

123    /**
124     * the listener will always be called, when the Properties-file changes.
125     *
126     * @param listener the listener
127     */
128    public void registerUpdateListener(Consumer<PropertiesAssistant> listener) {
129        if (listener != null)
130            listeners.add(new WeakReference<>(listener));
131    }

144     * Gets the path to default properties file path (the file which is copied into the real properties on start)

148    public String getDefaultPropertiesPath() {
149        return defaultPropertiesPath;
150    }

152    /**
153     * Initializes properties in the addOn. Creates new properties file using default properties.
154     */
155    public void initProperties() {
156        propertiesPath = getContext().getFiles().getPropertiesLocation() + File.separator
157                + getContext().getAddOn().getID() + ".properties";
158
159        this.propertiesFile = new File(propertiesPath);
160        if (!this.propertiesFile.exists()) try {
161            this.propertiesFile.createNewFile();
162        } catch (IOException e) {
163            error("Error while trying to create the new Properties file", e);
164        }
165
166        try {
167            BufferedReader in = new BufferedReader(new InputStreamReader(new FileInputStream(this.propertiesFile),
168                    "UTF8"));
169            try {
170                properties.load(in);
171            } catch (IOException e) {
172                error("unable to load the InputStream for the PropertiesFile",e);
173            }
174        } catch (FileNotFoundException | UnsupportedEncodingException e) {
175            error("Error while trying to read Properties-File", e);
176        }
177
178        if (defaultPropertiesPath != null && new File(defaultPropertiesPath).exists()) {
179            @SuppressWarnings("unchecked")
180            Enumeration<String> keys = (Enumeration<String>)properties.propertyNames();
181
182            if (!keys.hasMoreElements()) {
183                try {
184                    createDefaultPropertyFile(defaultPropertiesPath);
185                } catch (IOException e) {
186                    error("Error while trying to copy the Default-Properties File", e);
187                }
188
189                if (new File(defaultPropertiesPath).exists() && !writeToPropertiesFile(defaultPropertiesPath)) return;
190                reloadProperties();
191            }
192        }
193    }
194
195    /**
196     * Writes defaultPropertiesFile.txt to real properties file
197     * This is done so that the final user never has to worry about property file initialization
198     *
199     * @param defaultPropsPath path to defaultPropertyFile.txt (or where it should be created)
200     * @return true if operation has succeeded, else false
201     */
202    private boolean writeToPropertiesFile(String defaultPropsPath) {
203        return getContext().getFiles().writeToFile(defaultPropsPath, propertiesPath);
204    }
205
206    /**
207     * Creates a defaultPropertyFile.txt in case it does not exist yet. In case it is used by an addOn,
208     * it copies its content into the real properties file every time the addOn is launched.
209     *
210     * It is impossible to get the properties file on default, that way the user should not have to worry about
211     * the property file's initial content.
212     *
213     * @param defaultPropsPath path to defaultPropertyFile.txt (or where it should be created)
214     * @throws java.io.IOException is thrown by bufferedWriter
215     */
216    private void createDefaultPropertyFile(String defaultPropsPath) throws IOException {
217        getContext().getFiles().createDefaultFile(defaultPropsPath, "# Properties should always be in the "
218                + "form of: \"key = value\"");
219    }
220
221    /**
222     * reloads the propertiesFile into the propertiesContainer
223     */
224    private void reloadProperties() {
225        Properties temp = new Properties();
226        BufferedReader bufferedReader = null;
227        try {
228            File properties = new File(propertiesPath);
229            bufferedReader = new BufferedReader(new InputStreamReader(new FileInputStream(properties), "UTF8"));
230            temp.load(bufferedReader);
231            this.properties = temp;
232            listeners.removeIf(weakReference -> weakReference.get() == null);
233            listeners.forEach(weakReference -> {
234                Consumer<PropertiesAssistant> consumer = weakReference.get();
235                if (consumer != null)
236                    consumer.accept(this);
237            });
238        } catch (IOException e) {
239            error("Error while trying to load the Properties-File: "
240                    + propertiesPath, e);
241        } finally {
242            if (bufferedReader != null) {
243                try {
244                    bufferedReader.close();
245                } catch (IOException e) {
246                    error("Unable to close input stream", e);
247                }
248            }
249        }
250    }
251
252    @Override
253    public void reloadFile(String eventType) {
254        reloadProperties();
255    }
