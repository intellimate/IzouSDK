001package org.intellimate.izou.sdk.frameworks.presence.events;
002
003import org.intellimate.izou.identification.Identification;
004import org.intellimate.izou.sdk.events.CommonEvents;
005import org.intellimate.izou.sdk.events.Event;
006
007import java.util.List;
008import java.util.Optional;
009
010/**
011 * this event indicates presence
012 * @author LeanderK
013 * @version 1.0
014 */
015public class PresenceEvent extends Event {
016    /**
017     * this event does not mean the user is able to notice anything (can be used for warm-up), it indicates
018     * he might be
019     */
020    public static final String GENERAL_DESCRIPTOR = "izou.presence.general";
021    /**
022     * it means that the addon can guarantee that the user entered an area near izou
023     */
024    public static final String STRICT_DESCRIPTOR = "izou.presence.strict";
025    /**
026     * it means that someone entered an area near izou, but it is not clear if it is the user,
027     */
028    public static final String UNKNOWN_DESCRIPTOR = "izou.presence.unkownuser";
029    /**
030     * it means that the user entered an area near izou
031     */
032    public static final String KNOWN_DESCRIPTOR = "izou.presence.knownuser";
033    /**
034     * Means that the Addon encountered the user the first time IN THE SPECIFIED MODE (Strict/General) since he left.
035     */
036    public static final String FIRST_ENCOUNTER_DESCRIPTOR = "izou.presence.knownuser";
037    /**
038     * it means that the addon can guarantee that the user entered an area near izou
039     */
040    public static final String ID = "izou.presence";
041
042
043    /**
044     * Creates a new Event Object
045     *
046     * @param source      the source of the Event, most likely a this reference.
047     * @param descriptors the descriptors to initialize the Event with
048     * @throws IllegalArgumentException if one of the Arguments is null or empty
049     */
050    protected PresenceEvent(Identification source, List<String> descriptors) throws IllegalArgumentException {
051        super(CommonEvents.Type.RESPONSE_TYPE, source, descriptors);
052    }
053
054    /**
055     * creates a new PresenceEvent
056     * @param source the caller
057     * @param strict whether the addon can guarantee that the user is around
058     * @param known true if the addon knows if the person encountered is the (main) user
059     * @param firstEncounter if the user was first encountered in the specified mode
060     * @param descriptors the descriptors
061     * @return the optional PresenceEvent
062     */
063    public static Optional<PresenceEvent> createPresenceEvent(Identification source, boolean strict, boolean known,
064                                                                    boolean firstEncounter, List<String> descriptors) {
065        try {
066            if (strict) {
067                descriptors.add(STRICT_DESCRIPTOR);
068            } else {
069                descriptors.add(GENERAL_DESCRIPTOR);
070            }
071            if (known) {
072                descriptors.add(KNOWN_DESCRIPTOR);
073            } else {
074                descriptors.add(UNKNOWN_DESCRIPTOR);
075            }
076            if (firstEncounter)
077                descriptors.add(FIRST_ENCOUNTER_DESCRIPTOR);
078            descriptors.add(ID);
079            PresenceEvent stopRequest = new PresenceEvent(source, descriptors);
080            return Optional.of(stopRequest);
081        } catch (IllegalArgumentException e) {
082            return Optional.empty();
083        }
084    }
085}
