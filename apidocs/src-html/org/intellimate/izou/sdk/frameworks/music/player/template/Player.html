031/**
032 * use this class to actually play music.
033 * <p>
034 * <b>How to use it:</b><br>
035 * Usage is pretty straightforward. Extend this class an override the abstract methods.
036 * To register further commands, use the getCommandHandler() method in the constructor.
037 * to start playing, use the PlayerController.startPlaying() methods, never call the play() method directly!.
038 * Also, don't call the stopSound() method directly!.
039 * All basic methods are implemented in this class,but to code special behaviour fell free to explore the other
040 * classes and interfaces.<br>
041 * Note: the playback automatically stops when the user left.
042 * </p>
043 * @author LeanderK
044 * @version 1.0
045 */
046public abstract class Player<T> extends OutputPlugin<T> implements MusicProvider, ResourceBuilderModel, MusicHelper, EventListenerModel {

061    /**
062     * creates a new output-plugin with a new id
063     *  @param context context
064     * @param id      the id of the new output-plugin. The ID of the InformationProvider is also based on this id
065     *                (id + ".InformationProvider")
066     * @param runsInPlay whether the termination of the play method should be treated as the termination the
067     *                   music (STOP not PAUSE)
068     * @param activators the activators which are able to start the Player if the Player is not able to start from
069     *                   request from other addons
070     * @param providesTrackInfo whether the Player provides TrackInfo
071     * @param playbackShuffle whether the player is able to provide the info that the playback is shuffling
072     * @param playbackRepeat whether the player is able to provide the info that the playback is repeating
073     * @param playbackRepeatSong whether the player is able to provide the info that the playback is repeating the song
074     * @param isUsingJava true if using java, false if not (and for example a C-library)
075     */
076    public Player(Context context, String id, boolean runsInPlay, List<Identifiable> activators,
077                  boolean providesTrackInfo, boolean playbackShuffle, boolean playbackRepeat,
078                  boolean playbackRepeatSong, boolean isUsingJava) {
079        super(context, id);
080        this.runsInPlay = runsInPlay;
081        this.activators = activators;
082        this.isUsingJava = isUsingJava;
083        capabilities = new Capabilities(context);
084        if (providesTrackInfo)
085            capabilities.setProvidesTrackInfo(true);
086        if (playbackShuffle)
087            capabilities.setPlaybackShuffle(true);
088        if (playbackRepeat)
089            capabilities.setPlaybackRepeat(true);
090        if (playbackRepeatSong)
091            capabilities.setPlaybackRepeatSong(true);
092        commandHandler = createCommandHandler();
093        informationProvider = new InformationProvider(getContext(), getID(), this,
094                commandHandler);
095        getContext().getEvents().registerEventListener(Collections.singletonList(LeavingEvent.ID), this);
096    }

201    /**
202     * this method has no effect if runsInPlay is enabled in the constructor.
203     * stops the playing
204     */
205    public void stopMusicPlayback() {
206        if (runsInPlay || !isPlaying)
207            return;
208        isPlaying = false;
209        stopSound();
210        endedSound();
211        rollBackToDefault();
212        super.stop();
213    }

236    /**
237     * checks if the trackInfo is an update and fires the appropriate Event. This method should not be called
238     * without an active playlist, or an NullPointerException will be thrown.
239     * this method:<br>
240     *     - fires nothing if the trackInfo equals the current trackInfo.<br>
241     *     - fires an trackInfoUpdate if the trackInfo contains information not found in the current.<br>
242     */
243    @SuppressWarnings("unused")
244    public void updateCurrentTrackInfo(TrackInfo trackInfo) {
245        if (playlist.getCurrent().equals(trackInfo) ||
246                playlist.getCurrent().isNew(trackInfo))
247            return;
248        this.playlist = playlist.update(playlist.getCurrent(), trackInfo);
249        trackInfoUpdate(playlist, trackInfo);
250    }

257     */
258    @SuppressWarnings("unused")
259    public void trackInfoUpdate(Playlist playlist, TrackInfo info) {
260        this.playlist = playlist;

429    /**
430     * method that uses the data from the OutputExtensions to generate a final output that will then be rendered.
431     *
432     * @param data       the data generated
433     * @param eventModel the Event which caused the whole thing
434     */
435    @Override
436    public void renderFinalOutput(List<T> data, EventModel eventModel) {
437        Consumer<Runnable> checkOrCall = runnable -> {
438            List<ResourceModel> resourceModels =
439                    eventModel.getListResourceContainer().provideResource(SelectorResource.RESOURCE_ID);
440            if (resourceModels.isEmpty()) {
441                runnable.run();
442            } else {
443                resourceModels.stream()
444                        .map(resourceModel -> resourceModel.getResource() instanceof Identification ?
445                                ((Identification) resourceModel.getResource()) : null)
446                        .filter(Objects::nonNull)
447                        .filter(this::isOwner)
448                        .findAny()
449                        .ifPresent(id -> runnable.run());
450            }
451        };
452        if (StartMusicRequest.verify(eventModel, capabilities, this, activators)) {
453            if (isOutputRunning()) {
454                playerError(PlayerError.ERROR_ALREADY_PLAYING);
455                return;
456            }
457            playingThread = submit((Runnable) () -> {
458                //noinspection RedundantIfStatement
459                if (runsInPlay) {
460                    isRunning = false;
461                } else {
462                    isRunning = true;
463                }
464                isPlaying = true;
465                fireStartMusicRequest(eventModel);
466            }).thenRun(() -> play(eventModel))
467                    .thenRun(() -> {
468                        if (runsInPlay) {
469                            isRunning = false;
470                            isPlaying = false;
471                            endedSound();
472                        }
473                    });
474        }
475        if (eventModel.containsDescriptor(MuteEvent.ID)) {
476            checkOrCall.accept(this::mute);
477        }
478        if (eventModel.containsDescriptor(UnMuteEvent.ID)) {
479            checkOrCall.accept(this::unMute);
480        }
481        if (eventModel.containsDescriptor(StopEvent.ID)) {
482            checkOrCall.accept(this::stopMusicPlayback);
483        }
484        if (StopMusic.verify(eventModel, this)) {
485            stopMusicPlayback();
486        }
487        if (PlayerCommand.verify(eventModel, this)) {
488            getCommandHandler().handleCommandResources(eventModel);
489        }
490    }

500    /**
501     * this method will be called to create and fire the StartMusicRequest
502     * @param eventModel the cause
503     */
504    protected void fireStartMusicRequest(EventModel eventModel) {
505        Optional<Playlist> playlist = PlaylistResource.getPlaylist(eventModel);
506        Optional<Progress> progress = ProgressResource.getProgress(eventModel);
507        Optional<TrackInfo> trackInfo = TrackInfoResource.getTrackInfo(eventModel);
508        Optional<Volume> volume = VolumeResource.getVolume(eventModel);
509        startedSound(playlist.orElse(null), progress.orElse(null), trackInfo.orElse(null), volume.orElse(null), isUsingJava);
510    }

523    /**
524     * sets every information into its default state (playlist, volume, etc...)
525     */
526    public void rollBackToDefault() {
527        playlist = new Playlist(new ArrayList<>());
528        volume = Volume.createVolume(50).orElse(null);
529        progress = new Progress(0,0);
530        playbackState = PlaybackState.PLAY;
531        if (playingThread != null)
532            playingThread.cancel(true);
533        isRunning = false;
534        isPlaying = false;
535    }
