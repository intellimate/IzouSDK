001package org.intellimate.izou.sdk.frameworks.music.user;
002
003import org.intellimate.izou.identification.Identifiable;
004import org.intellimate.izou.identification.Identification;
005import org.intellimate.izou.resource.ResourceModel;
006import org.intellimate.izou.sdk.*;
007import org.intellimate.izou.sdk.frameworks.music.*;
008import org.intellimate.izou.sdk.frameworks.music.player.*;
009import org.intellimate.izou.sdk.frameworks.music.resources.*;
010import org.intellimate.izou.sdk.util.*;
011
012import java.util.ArrayList;
013import java.util.List;
014import java.util.Optional;
015import java.util.concurrent.CompletableFuture;
016import java.util.concurrent.ExecutionException;
017import java.util.concurrent.TimeUnit;
018import java.util.concurrent.TimeoutException;
019import java.util.function.Function;
020
021/**
022 * this class is used to get the available Playlists from the player. It can then build for example the PlayerRequest.
023 * <p>
024 * To use this class the player has to meet a few criteria:<br>
025 * <ul>
026 *  <li>the player must exist and be support the standard defined through the sdk</li>
027 *  <li>the players-capabilities must allow requests from outside</li>
028 *  <li>the players-capabilities must allow a requests with specified a specified playlist</li>
029 *  <li>the players-capabilities signal its broadcasting playlists</li>
030 * </ul><br>
031 * an example:<br>
032 * <pre>
033 * {@code
034 *
035 * Identification playerID;
036 * List<ResourceModel> resources = PlaylistSelector.getPlaylistsFromPlayer(playerID, this) //create a new PlaylistSelector
037 *              .flatMap(playlistSelector -> playlistSelector.getPlayerRequest("jazz")) //maps to a PlayerRequest which should hold the playlist jazz
038 *              .map(PlayerRequest::resourcesForExisting) //creates the resources needed to add to the event
039 *              .orElseGet(ArrayList::new);
040 * }
041 * </pre>
042 * @author LeanderK
043 * @version 1.0
044 */
045public class PlaylistSelector {
046    private final Identification player;
047    private final Capabilities capabilities;
048    private final List<String> playlists;
049    private final Context context;
050    private final Identifiable identifiable;
051
052    /**
053     * creates a new PlaylistSelector
054     * @param player the player
055     * @param capabilities the capabilities
056     * @param playlists the playlists
057     * @param context the context
058     * @param identifiable
059     */
060    protected PlaylistSelector(Identification player, Capabilities capabilities, List<String> playlists, Context context, Identifiable identifiable) {
061        this.player = player;
062        this.capabilities = capabilities;
063        this.playlists = playlists;
064        this.context = context;
065        this.identifiable = identifiable;
066    }
067
068    /**
069     * provides all the names of the broadcasted playlists
070     * @return a List with all the names
071     */
072    public List<String> getAvailablePlaylists() {
073        return playlists;
074    }
075
076    /**
077     * provides the ID of the associated player
078     * @return the player
079     */
080    public Identification getPlayer() {
081        return player;
082    }
083
084    /**
085     * returns the Capabilities of the Player
086     * @return the Capabilites
087     */
088    public Capabilities getCapabilities() {
089        return capabilities;
090    }
091
092    /**
093     * asks the Player for more information about the playlist.
094     * <p>
095     * This method may block for 1 second.
096     * @param playlistName the name of the playlist
097     * @return an optional answer of the player
098     */
099    public Optional<Playlist> getPlaylist(String playlistName) {
100        try {
101            return context.getResources().generateResource(BroadcasterPlaylist.createPlaylistRequest(player, playlistName))
102                    .orElse(CompletableFuture.completedFuture(new ArrayList<>()))
103                    .thenApply(list -> list.stream()
104                            .filter(resourceModel -> resourceModel.getProvider().equals(player))
105                            .findAny()
106                            .flatMap(Playlist::importResource)
107                    ).get(1, TimeUnit.SECONDS);
108        } catch (InterruptedException | ExecutionException | TimeoutException e) {
109            context.getLogger().error("unable to get Playlist");
110            return Optional.empty();
111        }
112    }
113
114    /**
115     * asks the player for more information about the specified playlist and creates a PlayerRequest with the answer.
116     * <p>
117     * the resulting PlayerRequest is not permanent, which means that it will mute all other sound but is limited to
118     * 10 minutes.
119     * @param playlistName the name of the Playlist
120     * @return an optional PlayerRequest
121     */
122    public Optional<PlayerRequest> getPlayerRequest(String playlistName) {
123        return getPlayerRequest(playlistName, false);
124    }
125
126    /**
127     * asks the player for more information about the specified playlist and creates a PlayerRequest with the answer.
128     * @param playlistName the name of the Playlist
129     * @param permanent true means the player can play indefinitely, but only if no one is currently using audio as
130     *                  permanent. It will also not block. false is limited to 10 minutes playback, but will block.
131     * @return the optional PlayerRequest
132     */
133    public Optional<PlayerRequest> getPlayerRequest(String playlistName, boolean permanent) {
134        return getPlaylist(playlistName)
135                .map(playlist -> PlayerRequest.createPlayerRequest(playlist, permanent, player, capabilities, context, identifiable));
136    }
137
138    /**
139     * creates a new PlaylistSelector.
140     * <p>
141     * For this method to return a non-empty Optional the following criteria must be met:<br>
142     * <ul>
143     *  <li>the player must exist</li>
144     *  <li>the players-capabilities must allow requests from outside</li>
145     *  <li>the players-capabilities must allow a requests with specified a specified playlist</li>
146     *  <li>the players-capabilities signal its broadcasting playlists</li>
147     * </ul>
148     * @param player the specified player
149     * @param addOnModule the addOnModule used for Context etc.
150     * @return the optional PlaylistSelector.
151     */
152    @SuppressWarnings("unused")
153    public static Optional<PlaylistSelector> getPlaylistsFromPlayer(Identification player, AddOnModule addOnModule) {
154        if (player == null || addOnModule == null)
155            return Optional.empty();
156        Function<Capabilities, CompletableFuture<Optional<PlaylistSelector>>> getPlaylistSelector = capabilities ->
157                addOnModule.getContext().getResources()
158                .generateResource(new BroadcasterAvailablePlaylists(player))
159                .orElse(CompletableFuture.completedFuture(new ArrayList<>()))
160                .thenApply(list -> list.stream()
161                                .filter(resourceModel -> resourceModel.getProvider().equals(player))
162                                .findAny()
163                                .flatMap(BroadcasterAvailablePlaylists::getPlaylists)
164                                .map(playlists ->
165                                        new PlaylistSelector(player, capabilities, playlists, addOnModule.getContext(), addOnModule))
166                );
167
168
169        Function<List<ResourceModel>, Optional<Capabilities>> getCapabilities = resourceModels -> resourceModels.stream()
170                .filter(resourceModel -> resourceModel.getProvider().equals(player))
171                .findAny()
172                .flatMap(resource -> Capabilities.importFromResource(resource, addOnModule.getContext()))
173                .filter(capabilities -> {
174                    if (!capabilities.handlesPlayRequestFromOutside()) {
175                        addOnModule.getContext().getLogger().error("player does not handle play-request from outside");
176                        return false;
177                    }
178                    if (!capabilities.hasPlayRequestDetailed()) {
179                        addOnModule.getContext().getLogger().error("player does not handle trackInfo-request from outside");
180                        return false;
181                    }
182                    if (!capabilities.isBroadcasting()) {
183                        addOnModule.getContext().getLogger().error("player is not broadcasting playlists");
184                        return false;
185                    }
186                    return true;
187                });
188
189        try {
190            return addOnModule.getContext().getResources()
191                    .generateResource(new CapabilitiesResource(player))
192                    .orElse(CompletableFuture.completedFuture(new ArrayList<>()))
193                    .thenApply(getCapabilities::apply)
194                    .thenCompose(capabilities -> capabilities.map(getPlaylistSelector::apply)
195                            .orElseGet(() -> CompletableFuture.completedFuture(Optional.<PlaylistSelector>empty()))
196                    ).get(1, TimeUnit.SECONDS);
197        } catch (InterruptedException | ExecutionException | TimeoutException e) {
198            addOnModule.getContext().getLogger().error("unable to get PlaylistSelector");
199            return Optional.empty();
200        }
201    }
202}
