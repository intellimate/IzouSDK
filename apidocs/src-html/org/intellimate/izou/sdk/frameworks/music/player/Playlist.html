001package org.intellimate.izou.sdk.frameworks.music.player;
002
003import org.intellimate.izou.resource.ResourceModel;
004import org.intellimate.izou.sdk.frameworks.music.Capabilities;

006import java.util.*;
007
008/**
009 * a playlist is a list of TrackInfo-objects, it is immutable!
010 * @author LeanderK
011 * @version 1.0
012 */
013@SuppressWarnings("unused")
014public class Playlist {
015    public static final String QUEUE_DESCRIPTOR = "izou.music.playlist.queue";
016    private final List<TrackInfo> queue;
017    public static final String POSITION_DESCRIPTOR = "izou.music.playlist.position";
018    private final int position;
019    public static final String NAME_DESCRIPTOR = "izou.music.playlist.name";
020    private final String name;
021    public static final String PLAYBACK_MODE_DESCRIPTOR = "izou.music.playlist.playbackmode";
022    private final List<PlaybackMode> playbackModes;
023    public static final String DATA_DESCRIPTOR = "izou.music.playlist.data";
024    private final String data;
025
026    public Playlist(String name) {
027        this(Collections.emptyList(), name, Collections.emptyList(), 0, null);
028    }
029
030    public Playlist(List<TrackInfo> queue) {
031        this(queue, null, Collections.emptyList(), 0, null);
032    }
033
034    public Playlist(List<TrackInfo> queue, String name, List<PlaybackMode> playbackModes, int position) {
035        this(queue, name, playbackModes, position, null);
036    }
037
038    public Playlist(List<TrackInfo> queue, String name, List<PlaybackMode> playbackModes, int position, String data) {
039        this.queue = Collections.unmodifiableList(new ArrayList<>(queue));
040        this.position = position;
041        this.name = name;
042        this.playbackModes = new ArrayList<>(playbackModes);
043        this.data = data;
044    }
045
046    /**
047     * returns the queue
048     * @return the List of TrackInfo-objects
049     */
050    public List<TrackInfo> getQueue() {
051        return queue;
052    }
053
054    /**
055     * returns the current played track or null
056     * @return an instance of TrackInfo
057     */
058    public TrackInfo getCurrent() {
059        TrackInfo trackInfo = null;
060        try {
061            trackInfo = queue.get(position);
062        } catch (IndexOutOfBoundsException e) {
063            trackInfo = null;
064        }
065        return trackInfo;
066    }
067
068    /**
069     * returns the current position
070     * @return the current position
071     */
072    public int getPosition() {
073        return position;
074    }
075
076    /**
077     * gets the name (optional)
078     * <p>
079     *     playlist doesn't need to have a name
080     * </p>
081     * @return the optional name
082     */
083    public Optional<String> getName() {
084        if (name == null) {
085            return Optional.empty();
086        } else {
087            return Optional.of(name);
088        }
089    }
090
091    /**
092     * retruns the active PlaybackModes
093     * @return the playbackModes
094     */
095    public List<PlaybackMode> getPlaybackModes() {
096        return playbackModes;
097    }
098
099    /**
100     *returns the associated data (not really specified, specified by implementation)
101     * @return the optional data-element
102     */
103    public Optional<String> getData() {
104        if (data == null) {
105            return Optional.empty();
106        } else {
107            return Optional.of(data);
108        }
109    }
110
111    @Override
112    public boolean equals(Object o) {
113        if (this == o) return true;
114        if (!(o instanceof Playlist)) return false;
115
116        Playlist playlist = (Playlist) o;
117
118        if (position != playlist.position) return false;
119        if (queue != null ? !queue.equals(playlist.queue) : playlist.queue != null) return false;
120        if (name != null ? !name.equals(playlist.name) : playlist.name != null) return false;
121        if (data != null ? !data.equals(playlist.data) : playlist.data != null) return false;
122        return !(playbackModes != null ? !playbackModes.equals(playlist.playbackModes) : playlist.playbackModes != null);
123
124    }
125
126    @Override
127    public int hashCode() {
128        int result = queue != null ? queue.hashCode() : 0;
129        result = 31 * result + position;
130        result = 31 * result + (name != null ? name.hashCode() : 0);
131        result = 31 * result + (playbackModes != null ? playbackModes.hashCode() : 0);
132        return result;
133    }
134
135    /**
136     * creates a new Playlist, with the next Track as head (or an illegal position if it is the last!!)
137     * @return a new Playlist
138     */
139    public Playlist nextTrack() {
140        return new Playlist(queue, name, playbackModes, position + 1);
141    }
142
143    /**
144     * creates a new Playlist, with the specified position
145     * @param position the position to set to
146     * @return a new Playlist
147     */
148    public Playlist setNewPosition(int position) {
149        return new Playlist(queue, name, playbackModes, position);
150    }
151
152    /**
153     * updates the TrackInfo-Object
154     * @param old the old TrackInfo
155     * @param newTrackInfo the new TrackInfo
156     * @return the resulting new Playlist
157     */
158    public Playlist update(TrackInfo old, TrackInfo newTrackInfo) {
159        List<TrackInfo> list = new ArrayList<>(queue);
160        list.set(list.indexOf(old), newTrackInfo);
161        return new Playlist(queue, name, playbackModes, position);
162    }
163
164    /**
165     * Shuffles the playlist and returns the shuffled playlist, so the original stays intact.
166     * Only the part of the playlist after the current position is shuffled.
167     *
168     * @return a shuffled copy of this playlist
169     */
170    public Playlist shuffle() {
171        int position = getPosition();
172        long seed = System.nanoTime();
173
174        if (position >= 0 && position < queue.size()) {
175            List<TrackInfo> trackInfos = queue.subList(0, position);
176            List<TrackInfo> notPlayed = queue.subList(position, queue.size());
177
178            List<TrackInfo> shuffledNotPlayed = new ArrayList<>(notPlayed);
179            Collections.shuffle(shuffledNotPlayed, new Random(seed));
180            trackInfos.addAll(shuffledNotPlayed);
181            return new Playlist(trackInfos);
182        } else {
183            List<TrackInfo> trackInfos = new ArrayList<>(queue);
184            Collections.shuffle(trackInfos, new Random(seed));
185            return new Playlist(trackInfos);
186        }
187    }
188
189    /**
190     * returns true if all the active playbackModes are supported
191     * @param capabilities the Capabilities to match against
192     * @return true if the player is capable, false if not
193     */
194    public boolean verify(Capabilities capabilities) {
195        for (PlaybackMode playbackMode : playbackModes) {
196            switch (playbackMode) {
197                case REPEAT:
198                    if (!capabilities.canRepeatPlayback()) {
199                        return false;
200                    } else {
201                        break;
202                    }
203                case REPEAT_SONG:
204                    if (!capabilities.canRepeatPlaybackOfSong()) {
205                        return false;
206                    } else {
207                        break;
208                    }
209                case SHUFFLE:
210                    if (!capabilities.canShufflePlayback()) {
211                        return false;
212                    } else {
213                        break;
214                    }
215            }
216        }
217        return true;
218    }
219
220    /**
221     * exports the Playlist to a HashMap
222     * @return a HashMap
223     */
224    public HashMap<String, Object> export() {
225        HashMap<String, Object> data = new HashMap<>();
226        for (int i = 0; i < queue.size(); i++) {
227            data.put(QUEUE_DESCRIPTOR+i, queue.get(i).export());
228        }
229        for (int i = 0; i < playbackModes.size(); i++) {
230            data.put(PLAYBACK_MODE_DESCRIPTOR+i, playbackModes.get(i).name());
231        }
232        data.put(NAME_DESCRIPTOR, name);
233        data.put(POSITION_DESCRIPTOR, position);
234        data.put(DATA_DESCRIPTOR, this.data);
235        return data;
236    }

238    /**
239     * constructs (if no errors were found) the Playlist from the Resource
240     * @param resourceModel the resourceModel to import from
241     * @return the optional Playlist
242     */
243    public static Optional<Playlist> importResource(ResourceModel resourceModel) {
244        Object resource = resourceModel.getResource();
245        try {
246            //noinspection unchecked
247            HashMap<String, Object> data = (HashMap<String, Object>) resource;
248            ArrayList<TrackInfo> queue = new ArrayList<>();
249            ArrayList<PlaybackMode> playbackModes = new ArrayList<>();
250            final String[] name = {null};
251            final int[] position = {-1};
252            final String[] dataString = {null};
253            data.entrySet().forEach(entry -> {
254                if (entry.getKey().startsWith(QUEUE_DESCRIPTOR)) {
255                    int index = Integer.parseInt(entry.getKey().replace(QUEUE_DESCRIPTOR, ""));
256                    //noinspection unchecked
257                    TrackInfo.importFromHashMap((HashMap<String, Object>) entry.getValue())
258                            .ifPresent(trackInfo -> queue.add(index, trackInfo));
259                } else if (entry.getKey().startsWith(PLAYBACK_MODE_DESCRIPTOR)) {
260                    try {
261                        int index = Integer.parseInt(entry.getKey().replace(PLAYBACK_MODE_DESCRIPTOR, ""));
262                        //noinspection unchecked
263                        playbackModes.add(index, PlaybackMode.valueOf((String) entry.getValue()));
264                    } catch (IllegalArgumentException ignored) {
265                        //happens when the name is not present...maybe future sdks define other playbackModes
266                    }
267                } else if (entry.getKey().equals(NAME_DESCRIPTOR)) {
268                    name[0] = (String) entry.getValue();
269                } else if (entry.getKey().equals(POSITION_DESCRIPTOR)) {
270                    position[0] = (int) entry.getValue();
271                } else if (entry.getKey().equals(DATA_DESCRIPTOR)) {
272                    dataString[0] = (String) entry.getValue();
273                }
274            });
275            return Optional.of(new Playlist(queue, name[0] , playbackModes, position[0], dataString[0]));
276        } catch (ClassCastException | IllegalArgumentException e) {
277            return Optional.empty();
278        }
279    }
