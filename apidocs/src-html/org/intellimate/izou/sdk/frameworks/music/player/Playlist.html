001package org.intellimate.izou.sdk.frameworks.music.player;
002
003import org.intellimate.izou.resource.ResourceModel;
004
005import java.util.*;
006
007/**
008 * a playlist is a list of TrackInfo-objects, it is immutable!
009 * @author LeanderK
010 * @version 1.0
011 */
012@SuppressWarnings("unused")
013public class Playlist {
014    public static final String QUEUE_DESCRIPTOR = "izou.music.playlist.queue";
015    private final List<TrackInfo> queue;
016    public static final String POSITION_DESCRIPTOR = "izou.music.playlist.position";
017    private final int position;
018    public static final String NAME_DESCRIPTOR = "izou.music.playlist.name";
019    private final String name;
020    public static final String PLAYBACK_MODE_DESCRIPTOR = "izou.music.playlist.playbackmode";
021    private final List<PlaybackMode> playbackModes;
022    public static final String DATA_DESCRIPTOR = "izou.music.playlist.data";
023    private final String data;
024
025    public Playlist(String name) {
026        this(Collections.emptyList(), name, Collections.emptyList(), 0, null);
027    }
028
029    public Playlist(List<TrackInfo> queue) {
030        this(queue, null, Collections.emptyList(), 0, null);
031    }
032
033    public Playlist(List<TrackInfo> queue, String name, List<PlaybackMode> playbackModes, int position) {
034        this(queue, name, playbackModes, position, null);
035    }
036
037    public Playlist(List<TrackInfo> queue, String name, List<PlaybackMode> playbackModes, int position, String data) {
038        this.queue = Collections.unmodifiableList(new ArrayList<>(queue));
039        this.position = position;
040        this.name = name;
041        this.playbackModes = new ArrayList<>(playbackModes);
042        this.data = data;
043    }
044
045    /**
046     * returns the queue
047     * @return the List of TrackInfo-objects
048     */
049    public List<TrackInfo> getQueue() {
050        return queue;
051    }
052
053    /**
054     * returns the current played track or null
055     * @return an instance of TrackInfo
056     */
057    public TrackInfo getCurrent() {
058        TrackInfo trackInfo = null;
059        try {
060            trackInfo = queue.get(position);
061        } catch (IndexOutOfBoundsException e) {
062            trackInfo = null;
063        }
064        return trackInfo;
065    }
066
067    /**
068     * returns the current position
069     * @return the current position
070     */
071    public int getPosition() {
072        return position;
073    }
074
075    /**
076     * gets the name (optional)
077     * <p>
078     *     playlist doesn't need to have a name
079     * </p>
080     * @return the optional name
081     */
082    public Optional<String> getName() {
083        if (name == null) {
084            return Optional.empty();
085        } else {
086            return Optional.of(name);
087        }
088    }
089
090    /**
091     * retruns the active PlaybackModes
092     * @return the playbackModes
093     */
094    public List<PlaybackMode> getPlaybackModes() {
095        return playbackModes;
096    }
097
098    /**
099     *returns the associated data (not really specified, specified by implementation)
100     * @return the optional data-element
101     */
102    public Optional<String> getData() {
103        if (data == null) {
104            return Optional.empty();
105        } else {
106            return Optional.of(data);
107        }
108    }
109
110    @Override
111    public boolean equals(Object o) {
112        if (this == o) return true;
113        if (!(o instanceof Playlist)) return false;
114
115        Playlist playlist = (Playlist) o;
116
117        if (position != playlist.position) return false;
118        if (queue != null ? !queue.equals(playlist.queue) : playlist.queue != null) return false;
119        if (name != null ? !name.equals(playlist.name) : playlist.name != null) return false;
120        if (data != null ? !data.equals(playlist.data) : playlist.data != null) return false;
121        return !(playbackModes != null ? !playbackModes.equals(playlist.playbackModes) : playlist.playbackModes != null);
122
123    }
124
125    @Override
126    public int hashCode() {
127        int result = queue != null ? queue.hashCode() : 0;
128        result = 31 * result + position;
129        result = 31 * result + (name != null ? name.hashCode() : 0);
130        result = 31 * result + (playbackModes != null ? playbackModes.hashCode() : 0);
131        return result;
132    }
133
134    /**
135     * creates a new Playlist, with the next Track as head (or an illegal position if it is the last!!)
136     * @return a new Playlist
137     */
138    public Playlist nextTrack() {
139        return new Playlist(queue, name, playbackModes, position + 1);
140    }
141
142    /**
143     * creates a new Playlist, with the specified position
144     * @param position the position to set to
145     * @return a new Playlist
146     */
147    public Playlist setNewPosition(int position) {
148        return new Playlist(queue, name, playbackModes, position);
149    }
150
151    /**
152     * updates the TrackInfo-Object
153     * @param old the old TrackInfo
154     * @param newTrackInfo the new TrackInfo
155     * @return the resulting new Playlist
156     */
157    public Playlist update(TrackInfo old, TrackInfo newTrackInfo) {
158        List<TrackInfo> list = new ArrayList<>(queue);
159        list.set(list.indexOf(old), newTrackInfo);
160        return new Playlist(queue, name, playbackModes, position);
161    }
162
163    /**
164     * Shuffles the playlist and returns the shuffled playlist, so the original stays intact.
165     * Only the part of the playlist after the current position is shuffled.
166     *
167     * @return a shuffled copy of this playlist
168     */
169    public Playlist shuffle() {
170        int position = getPosition();
171        long seed = System.nanoTime();
172
173        if (position >= 0 && position < queue.size()) {
174            List<TrackInfo> trackInfos = queue.subList(0, position);
175            List<TrackInfo> notPlayed = queue.subList(position, queue.size());
176
177            List<TrackInfo> shuffledNotPlayed = new ArrayList<>(notPlayed);
178            Collections.shuffle(shuffledNotPlayed, new Random(seed));
179            trackInfos.addAll(shuffledNotPlayed);
180            return new Playlist(trackInfos);
181        } else {
182            List<TrackInfo> trackInfos = new ArrayList<>(queue);
183            Collections.shuffle(trackInfos, new Random(seed));
184            return new Playlist(trackInfos);
185        }
186    }
187
188    /**
189     * exports the Playlist to a HashMap
190     * @return a HashMap
191     */
192    public HashMap<String, Object> export() {
193        HashMap<String, Object> data = new HashMap<>();
194        for (int i = 0; i < queue.size(); i++) {
195            data.put(QUEUE_DESCRIPTOR+i, queue.get(i).export());
196        }
197        for (int i = 0; i < playbackModes.size(); i++) {
198            data.put(PLAYBACK_MODE_DESCRIPTOR+i, playbackModes.get(i).name());
199        }
200        data.put(NAME_DESCRIPTOR, name);
201        data.put(POSITION_DESCRIPTOR, position);
202        data.put(DATA_DESCRIPTOR, this.data);
203        return data;
204    }
205
206    /**
207     * constructs (if no errors were found) the Playlist from the Resource
208     * @param resourceModel the resourceModel to import from
209     * @return the optional Playlist
210     */
211    public static Optional<Playlist> importResource(ResourceModel resourceModel) {
212        Object resource = resourceModel.getResource();
213        try {
214            //noinspection unchecked
215            HashMap<String, Object> data = (HashMap<String, Object>) resource;
216            ArrayList<TrackInfo> queue = new ArrayList<>();
217            ArrayList<PlaybackMode> playbackModes = new ArrayList<>();
218            final String[] name = {null};
219            final int[] position = {-1};
220            final String[] dataString = {null};
221            data.entrySet().forEach(entry -> {
222                if (entry.getKey().startsWith(QUEUE_DESCRIPTOR)) {
223                    int index = Integer.parseInt(entry.getKey().replace(QUEUE_DESCRIPTOR, ""));
224                    //noinspection unchecked
225                    TrackInfo.importFromHashMap((HashMap<String, Object>) entry.getValue())
226                            .ifPresent(trackInfo -> queue.add(index, trackInfo));
227                } else if (entry.getKey().startsWith(PLAYBACK_MODE_DESCRIPTOR)) {
228                    try {
229                        int index = Integer.parseInt(entry.getKey().replace(PLAYBACK_MODE_DESCRIPTOR, ""));
230                        //noinspection unchecked
231                        playbackModes.add(index, PlaybackMode.valueOf((String) entry.getValue()));
232                    } catch (IllegalArgumentException ignored) {
233                        //happens when the name is not present...maybe future sdks define other playbackModes
234                    }
235                } else if (entry.getKey().equals(NAME_DESCRIPTOR)) {
236                    name[0] = (String) entry.getValue();
237                } else if (entry.getKey().equals(POSITION_DESCRIPTOR)) {
238                    position[0] = (int) entry.getValue();
239                } else if (entry.getKey().equals(DATA_DESCRIPTOR)) {
240                    dataString[0] = (String) entry.getValue();
241                }
242            });
243            return Optional.of(new Playlist(queue, name[0] , playbackModes, position[0], dataString[0]));
244        } catch (ClassCastException | IllegalArgumentException e) {
245            return Optional.empty();
246        }
247    }
